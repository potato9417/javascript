for(var i=0;i<gnb.length;i++)
{
    var item = gnb.item(i);
    item.addEventListener("mouseenter",function(){
        header.classList.add("open");
    })
    item.addEventListener("mouseleave",function(){
        header.classList.remove("open");
    })
